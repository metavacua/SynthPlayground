# A "total function" witness for the Collatz conjecture.
# This program computes one step of the Collatz sequence.
# It is a total function because it takes a "fuel" parameter `k`
# and is guaranteed to terminate in at most `k` steps.
#
# Input: n, k (integer, integer)
# Output: n', k-1 (integer, integer) or HALT if k=0.

FUNCTION collatz_step(n, k):
  IF k == 0:
    RETURN HALT

  IF n % 2 == 0:
    n_prime = n / 2
  ELSE:
    n_prime = 3 * n + 1

  k_prime = k - 1

  RETURN n_prime, k_prime
