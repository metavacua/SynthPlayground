# `control_program.txt`
#
# A witness for the unbounded control logic of the Collatz conjecture.
# This program is not guaranteed to terminate.
# It invokes the `collatz_total` function and decides whether to continue.
#
# Input: n (integer)
# Output: HALT (SUCCESS) if the sequence reaches 1.

PROGRAM collatz_control(n):
  k = 100 # Initial fuel

  LOOP:
    n, k = CALL collatz_step(n, k)

    IF n == 1:
      RETURN HALT (SUCCESS)

    IF n == HALT:
      # Fuel exhausted. Decide whether to continue.
      # For this simple witness, we will just add more fuel.
      # A more complex control program could have more sophisticated logic.
      k = k + 100
      CONTINUE LOOP
